C# 类的学习

2020-12-17 #C#

✍️ C# 是一个现代的、通用的、面向对象的编程语言，它是由微软（Microsoft）开发的，由 Ecma 和 ISO 核准认可的。

定义一个类

//定义一个人员类
class Person {
 	//编写类的成员
}

类的成员和修饰词

类的成员有:字段，属性，方法，构造，索引器，

类的修饰有:公有，私有（保护），抽象

类的成员-字段

//定义一个人员类
class Person {
    //字段
    private  string name;
    private int age;
    private string sex;
    private static bool isMorf;
}

类的成员-属性

//定义一个人员类
class Person {
    //字段(上述代码)
     
    //Name
    public string Name {
        get { return name; }
        set { name = value; }
    }

    //Age
    public int Age {
        set { age = value; }
        private get { return age; }

    }

    //性别
    public string Sex{
        set { sex = value; }
        get { return sex; }
    }       

    //静态属性不能使用非静态字段
    public static bool Morf {
        get { return isMorf; }
        set { isMorf = value; }
    }
    
    
}

类的成员-方法

class Person {
    //方法
    public void Print() {
        Console.WriteLine("none params");
    }
}

类的成员-索引器

class Person {
    public int this[int index]
    {
        get { return intarray[index]; }
        set { intarray[index] = value; }
    }
}

类的成员-构造

class Person {
 	//实例构造方法
    public Person()
    {
        Console.WriteLine("实例构造函数调用了");
    }
    //静态构造方法
    static Person()
    {
        Console.WriteLine("静态构造函数调用了");
    }
    //析构构造函数
    ~Person()
    {
        Console.WriteLine("析构构造函数被调用");
    }

}

类的使用

static void Main(string[] args)
{
    //构造函数 实例化对象
    Person p = new Person();
    
    //属性赋值
    p.Name = "Tony";
    p.Age = 12;
    p.Sex = "f";

    //静态的方法只能类调用
    Person.Morf = false;

    //调用方法 
    p.Print();
    
    //索引器使用
    p[0] = 12;
    p[1] = 13;
    p[2] = 14;
    Console.WriteLine(" p[0]=" +  p[0]);
    Console.WriteLine(" p[1]=" +  p[1]);
    Console.WriteLine(" p[2]=" +  p[2]);


}
